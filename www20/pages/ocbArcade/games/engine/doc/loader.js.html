define("pwge/loader", function(){
    var imageMap = {},
        /**
         * loader 모듈
         * @exports pwge/loader
         */
        loader = {};

    /**
     * 이미지를 로드한다.
     * @param  {Array}   arr      로드할 이미지 url 배열
     * @param  {Function} callback 로드완료시 callback
     */
    loader.loadImages = function(arr, callback) {
        if (!Array.isArray(arr)) {
            this.loadImages.call(this, [arr], callback);
            return;
        }

        var done = 0,
            i, len = arr.length,
            img,
            images = [],
            imgOnload = (function() {
                done += 1;
                // done / len;
                if (done === len) {
                    callback.apply(this, images);
                }
            }).bind(this);

        for (i = 0; i < len; i++) {
            img = new Image();
            img.onload = imgOnload;
            img.src = arr[i];

            images.push(img);
            imageMap[encodeURIComponent(arr[i])] = img;
        }
    };

    /**
     * 로드된 이미지를 불러온다.
     * @param  {String} url 불러올 이미지의 url
     * @return {HTMLImageElement}     이미지 엘리먼트
     */
    loader.get = function(url) {
        return imageMap[encodeURIComponent(url)] || null;
    };

    return loader;
});
